(defun binary-search (L)
    (cond 
        ((null L) T)
        ((and (not (null (car (car (cdr L))))) (< (car L) (car (car (cdr L))))) nil)
        ((and (not (null (car (car (cdr (cdr L)))))) (> (car L) (car (car (cdr (cdr L)))))) nil)
        (t (and (binary-search (car (cdr L))) (binary-search (car (cdr L)))))))
